### perl script
	#!/usr/bin/perl -w
	use strict;
	use warnings;
	open(IN,"<",$ARGV[0]) or die $!;
	my $seq=<IN>;
	my $i=index($seq,"ATG");
	my $protein="";
	for (my $i;$i<length($seq);$i+=3){
        my condon=substr($seq,$i,3);
        $protein.=&trans($condon);
        last if(&trans($condon)eq "_");
	}
	print $protein;
	sub trans{
        my ($condon)=@_;
        my (%genetic_code)=('TCA' => 'S',    # Serine
